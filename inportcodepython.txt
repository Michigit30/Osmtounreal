import unreal
import json
import os

def import_splines_from_json(json_path, blueprint_path):
  # 1. Check if the file exists
    if not os.path.exists(json_path):
        unreal.log_error(f"Fehler: JSON-Datei nicht gefunden unter {json_path}")
        return

    # 2. Load JSON data
    try:
        with open(json_path, 'r', encoding='utf-8') as file:
            data = json.load(file)
    except Exception as e:
        unreal.log_error(f"Fehler beim Lesen der JSON-Datei: {e}")
        return
# 3. Load Blueprint class
    bp_class = unreal.EditorAssetLibrary.load_blueprint_class(blueprint_path)
    if not bp_class:
        unreal.log_error(f"Fehler: Konnte Blueprint '{blueprint_path}' nicht finden. Hast du BP_RoadSpline erstellt?")
        return

    # Editor Subsystem zum Spawnen von Actors holen
    editor_subsystem = unreal.get_editor_subsystem(unreal.EditorActorSubsystem)

    # 4. Iterate Data and Create Splines
    for road in data:
        road_key = road.get("key", "UnknownRoad")
        is_closed = road.get("bClosed", False)
        points = road.get("controlPointsCm", [])

        if not points:
            continue

        # Spawn actor at zero point
        spawn_location = unreal.Vector(0, 0, 0)
        spawn_rotation = unreal.Rotator(0, 0, 0)
        new_actor = editor_subsystem.spawn_actor_from_class(bp_class, spawn_location, spawn_rotation)

        if new_actor:
            # Setze den Namen des Actors im Outliner
            new_actor.set_actor_label(f"Spline_{road_key}")

            # Spline-Komponente aus dem gespawnten Blueprint holen
            spline_comp = new_actor.get_component_by_class(unreal.SplineComponent)

            if spline_comp:
                # Standard-Punkte der neuen Spline löschen
                spline_comp.clear_spline_points(True)

                # Add JSON points
                for i, point in enumerate(points):
                    # Koordinaten übernehmen (JSON ist bereits in Zentimetern, genau wie Unreal)
                    ue_vector = unreal.Vector(point["X"], point["Y"], point["Z"])

                    # Punkt im World-Space hinzufügen
                    spline_comp.add_spline_point(ue_vector, unreal.SplineCoordinateSpace.WORLD, True)

                    # Punkttyp auf Kurve (für weiche Übergänge) oder Linear (für harte Kanten) setzen
                    spline_comp.set_spline_point_type(i, unreal.SplinePointType.CURVE, True)

                # Falls es sich um einen Kreisverkehr handelt (bClosed = true), Spline schließen
                spline_comp.set_closed_loop(is_closed, True)

    unreal.log_warning(f"Erfolgreich {len(data)} Splines in das Level importiert!")

# ==========================================
# EXECUTION
# ==========================================

# IMPORTANT: Replace this path with the real path to your .json file on your PC!
# Use r"Path" (Raw String) because of the backslashes in Windows.
json_file_path = r"C:Path/ue-pcg-area-splines.json"

# The path to the Blueprint you created in step 2
blueprint_asset_path = "/Game/BP_RoadSpline"

# Start script
import_splines_from_json(json_file_path, blueprint_asset_path)
